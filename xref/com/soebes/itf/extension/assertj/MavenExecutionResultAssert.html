1   package com.soebes.itf.extension.assertj;
2   
3   import org.assertj.core.api.AbstractAssert;
4   
5   import com.soebes.itf.jupiter.maven.MavenExecutionResult;
6   
7   /**
8    * @author Karl Heinz Marbaise
9    */
10  public class MavenExecutionResultAssert extends AbstractAssert<MavenExecutionResultAssert, MavenExecutionResult> {
11  
12    protected MavenExecutionResultAssert(MavenExecutionResult actual) {
13      super(actual, MavenExecutionResultAssert.class);
14    }
15  
16    /**
17     * An entry point for MavenExecutionResultAssert to follow AssertJ standard <code>assertThat()</code> statements.<br>
18     * With a static import, one's can write directly : <code>assertThat(result).isSuccessful();</code>
19     *
20     * @param actual the MavenExecutionResult we want to make assertions on.
21     * @return a new </code>{@link MavenExecutionResultAssert}</code>
22     */
23    public static MavenExecutionResultAssert assertThat(MavenExecutionResult actual) {
24      return new MavenExecutionResultAssert(actual);
25    }
26  
27    public MavenLogAssert log() {
28      isNotNull();
29      return new MavenLogAssert(this.actual.getMavenLog());
30    }
31  
32    public MavenProjectResultAssert project() {
33      isNotNull();
34      return new MavenProjectResultAssert(this.actual.getMavenProjectResult());
35    }
36  
37    public MavenCacheResultAssert cache() {
38      isNotNull();
39      return new MavenCacheResultAssert(this.actual.getMavenCacheResult());
40    }
41    /**
42     * @return {@link MavenExecutionResultAssert} for method chaining.
43     * @throws AssertionError if the actual value is {@code null}.
44     * @throws AssertionError if the actual value is not {@code true}.
45     */
46    public MavenExecutionResultAssert isSuccessful() {
47      isNotNull();
48      if (!this.actual.isSuccesful()) {
49        failWithMessage("The build was not successful but was <%s> with returnCode:<%s>", actual.getResult(),
50            actual.getReturnCode());
51      }
52      return myself;
53    }
54    /**
55     * @return {@link MavenExecutionResultAssert} for method chaining.
56     * @throws AssertionError if the actual value is {@code null}.
57     * @throws AssertionError if the actual value is not {@code true}.
58     */
59    public MavenExecutionResultAssert isFailure() {
60      isNotNull();
61      if (!this.actual.isFailure()) {
62        failWithMessage("The build should be not successful but was <%s> with returnCode:<%s>", actual.getResult(),
63            actual.getReturnCode());
64      }
65      return myself;
66    }
67  }
