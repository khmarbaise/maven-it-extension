29  import java.lang.annotation.RetentionPolicy;
30  import java.lang.annotation.Target;

32  import static org.apiguardian.api.API.Status.EXPERIMENTAL;
33  
34  /**
35   * {@code @EnabledForMavenVersion} is used to signal that the annotated test class or test method is
36   * <em>EnabledForMavenVersion</em> on one or more specified Maven version which the tests are being run with.
37   *
38   * <p>When applied at the class level, all test methods within that class
39   * will be disabled on the same specified Maven version.
40   *
41   * <p>If a test method is disabled via this annotation, that does not prevent
42   * the test class from being instantiated. Rather, it prevents the execution of the test method and method-level
43   * lifecycle callbacks such as {@code @BeforeEach} methods, {@code @AfterEach} methods, and corresponding extension
44   * APIs.
45   *
46   * <p>This annotation may be used as a meta-annotation in order to create a
47   * custom <em>composed annotation</em> that inherits the semantics of this annotation.
48   *
49   * @author Karl Heinz Marbaise
50   * @see MavenVersion
51   * @see org.junit.jupiter.api.condition.EnabledOnOs
52   * @see org.junit.jupiter.api.condition.EnabledOnJre
53   * @see org.junit.jupiter.api.condition.DisabledOnJre
54   * @see org.junit.jupiter.api.condition.EnabledIfEnvironmentVariable
55   * @see org.junit.jupiter.api.condition.DisabledIfEnvironmentVariable
56   * @see org.junit.jupiter.api.condition.EnabledIfSystemProperty
57   * @see org.junit.jupiter.api.condition.DisabledIfSystemProperty
58   * @see org.junit.jupiter.api.Disabled
59   * @since 0.1
60   */
61  @Target({ElementType.TYPE, ElementType.METHOD})
62  @Retention(RetentionPolicy.RUNTIME)
63  @Documented
64  @ExtendWith(EnabledForMavenVersionCondition.class)
65  @API(status = EXPERIMENTAL, since = "0.1.0")
66  public @interface EnabledForMavenVersion {
67  
68    MavenVersion[] value();
69  }
